const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); function doGet(e) { const action = e.parameter.action || 'load'; if (action === 'load') { return jsonResponse({ customers: getCustomers(), monthly: getMonthly(), settings: getSettings(), transactions: getTransactions() }); } if (action === 'customers') { return jsonResponse(getCustomers()); } if (action === 'monthly') { return jsonResponse(getMonthly()); } return jsonResponse({ error: 'Unknown action' }, 400); } function doPost(e) { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'saveCustomers') { saveCustomers(data.customers); return jsonResponse({ success: true, saved: data.customers.length }); } if (action === 'addTransaction') { addTransaction(data.transaction); return jsonResponse({ success: true }); } if (action === 'updateSettings') { updateSettings(data.settings); return jsonResponse({ success: true }); } if (action === 'saveAll') { if (data.customers) saveCustomers(data.customers); if (data.monthly) saveMonthly(data.monthly); if (data.settings) updateSettings(data.settings); if (data.transactions) saveTransactions(data.transactions); return jsonResponse({ success: true }); } return jsonResponse({ error: 'Unknown action' }, 400); } function jsonResponse(data, statusCode) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); if (name === 'Customers') { sheet.getRange(1, 1, 1, 11).setValues([['Name','Contact','Spent','Visits','Days','FirstVisit','LastVisit','Masked','IsNew','Resolved','UpdatedAt']]); } if (name === 'MonthlyRevenue') { sheet.getRange(1, 1, 1, 2).setValues([['Month','Revenue']]); } if (name === 'Transactions') { sheet.getRange(1, 1, 1, 7).setValues([['Receipt','Date','Time','Name','Phone','Amount','ImportedAt']]); } if (name === 'Settings') { sheet.getRange(1, 1, 1, 2).setValues([['Key','Value']]); sheet.getRange(2, 1, 3, 2).setValues([['importedRev', '0'], ['importedTx', '0'], ['resolved', '0']]); } } return sheet; } function getCustomers() { const sheet = getSheet('Customers'); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const rows = data.slice(1); return rows.map(row => ({ name: row[0] || '', contact: row[1] || '', spent: Number(row[2]) || 0, visits: Number(row[3]) || 0, days: Number(row[4]) || 999, firstVisit: row[5] || '', lastVisit: row[6] || '', masked: row[7] === true || row[7] === 'TRUE' || row[7] === 'true', isNew: row[8] === true || row[8] === 'TRUE' || row[8] === 'true', resolved: row[9] === true || row[9] === 'TRUE' || row[9] === 'true', updatedAt: row[10] || '' })); } function saveCustomers(customers) { const sheet = getSheet('Customers'); if (sheet.getLastRow() > 1) { sheet.getRange(2, 1, sheet.getLastRow() - 1, 11).clear(); } if (customers.length === 0) return; const now = new Date().toISOString(); const rows = customers.map(c => [ c.name || '', c.contact || '', Number(c.spent) || 0, Number(c.visits) || 0, Number(c.days) || 999, c.firstVisit || '', c.lastVisit || '', c.masked ? 'TRUE' : 'FALSE', c.isNew ? 'TRUE' : 'FALSE', c.resolved ? 'TRUE' : 'FALSE', now ]); sheet.getRange(2, 1, rows.length, 11).setValues(rows); } function getMonthly() { const sheet = getSheet('MonthlyRevenue'); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { labels: [], revenue: [] }; const rows = data.slice(1); return { labels: rows.map(r => r[0]).filter(Boolean), revenue: rows.map(r => Number(r[1]) || 0) }; } function saveMonthly(monthly) { const sheet = getSheet('MonthlyRevenue'); if (sheet.getLastRow() > 1) { sheet.getRange(2, 1, sheet.getLastRow() - 1, 2).clear(); } if (!monthly.labels || monthly.labels.length === 0) return; const rows = monthly.labels.map((label, i) => [label, monthly.revenue[i] || 0]); sheet.getRange(2, 1, rows.length, 2).setValues(rows); } function getTransactions() { const sheet = getSheet('Transactions'); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const rows = data.slice(1); return rows.map(r => ({ receipt: r[0] || '', date: r[1] || '', time: r[2] || '', name: r[3] || '', phone: r[4] || '', amount: Number(r[5]) || 0, importedAt: r[6] || '' })); } function addTransaction(tx) { const sheet = getSheet('Transactions'); sheet.appendRow([ tx.receipt || '', tx.date || '', tx.time || '', tx.name || '', tx.phone || '', Number(tx.amount) || 0, new Date().toISOString() ]); } function saveTransactions(txs) { const sheet = getSheet('Transactions'); if (sheet.getLastRow() > 1) { sheet.getRange(2, 1, sheet.getLastRow() - 1, 7).clear(); } if (!txs || txs.length === 0) return; const rows = txs.map(tx => [ tx.receipt || '', tx.date || '', tx.time || '', tx.name || '', tx.phone || '', Number(tx.amount) || 0, tx.importedAt || new Date().toISOString() ]); sheet.getRange(2, 1, rows.length, 7).setValues(rows); } function getSettings() { const sheet = getSheet('Settings'); const data = sheet.getDataRange().getValues(); const settings = {}; for (let i = 1; i < data.length; i++) { settings[data[i][0]] = data[i][1]; } return settings; } function updateSettings(settings) { const sheet = getSheet('Settings'); const data = sheet.getDataRange().getValues(); const existingKeys = {}; for (let i = 1; i < data.length; i++) { existingKeys[data[i][0]] = i + 1; } Object.entries(settings).forEach(([key, value]) => { if (existingKeys[key]) { sheet.getRange(existingKeys[key], 2).setValue(value); } else { sheet.appendRow([key, value]); } }); } function testLoad() { Logger.log(JSON.stringify(getCustomers().slice(0, 3), null, 2));
